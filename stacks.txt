#include<bits/stdc++.h>

using namespace std;

const int sz=100000;

template<class T>
class Stack
{	
	private :
		T stk[sz+1];
		int top;
	public :
		Stack(){
			top=-1;
		}
		void push(T val);
		void pop();
		bool isEmpty();
		T peak();
		~Stack(){
			
		}
};

template<class T>
bool Stack<T>::isEmpty(){
	if(top==-1){
		return true;
	}
	else{
		return false;
	}
}

template<class T>
void Stack<T>::push(T val){
	if(top==sz){
		cout<<"Stack overflow\n";
	}
	else{
		top++;
		stk[top]=val;
	}
}

template<class T>
void Stack<T>::pop(){
	if(isEmpty()){
		cout<<"Stack underflow\n";
	}
	else{
		top--;
	}
}


template<class T>
T Stack<T>::peak(){
	if(!isEmpty()){
		return stk[top];
	}
	else{
		cout<<"Stack is empty\n";
		return 0;
	}
}
int main(){
	Stack <int>s ;
	Stack <char> s1;
	cout<<s.isEmpty()<<"\n";
	s.push(1);
	s.push(2);
	s.push(3);
	s.push(4);
	cout<<s.peak()<<"\n";
	s.pop();
	cout<<s.peak()<<"\n";
	cout<<s.isEmpty()<<"\n";
	s1.push('m');
	s1.push('a');
	s1.push('h');
	s1.push('i');
	cout<<s1.peak()<<"\n";
	s1.pop();
	cout<<s1.peak()<<"\n";
	s1.pop();
	cout<<s1.peak()<<"\n";
	s1.pop();
	cout<<s1.peak()<<"\n";
	s1.pop();
	return 0;
}
